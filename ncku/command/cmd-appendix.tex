% Some helper function about appendix

% ----------------------------------------------------------------------------

\def \StartAppendix
{
  \StartNewPage
  % Appendix begins here (For one appendix chapter only)
  %\cleardoublepage
  \appendix
  %\clearpage
  %\setcounter{page}{\thepage}
  % If more than one appendix chapters
  %\appendices
}

\def \EndAppendix {\EndOfPage}
